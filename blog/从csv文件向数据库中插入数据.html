import csv

file_path = "I:/db/flight4.csv"

创建数据库

con = sqlite3.connect("I:/db/flight.db")
# con.execute("drop table flight_info")
# con.commit()

c = con.cursor()
create_table = """
        create table flight_info
        (sno varchar(20),
        start_time date,
        end_time date,
        start_p varchar(15),
        end_p varchar(15),
        money int,
        remaining int,
        time varchar(10)
        );
"""
c.execute(create_table)
con.commit()

读取数据到数据库中
